.data
ZERO: DB #141, #0
COLOR: DB #142, #255
WIDTH: DB #143, #5
HEIGHT: DB #144, #3
Y: DB #145, #0
X: DB #146, #0
RAND1: DB #147, #0
RAND2: DB #148, #0
RAND3: DB #149, #0
.enddata

.code
START:	LD Y
	SUB HEIGHT
	JZ END
DRAW:	LD ZERO
	LD2 COLOR
	RND RAND1
	RND RAND2
	RND RAND3	
	LD X
	LD2 Y
	POS
	LD RAND1
	LD2 RAND2
	LD3 RAND3
	PXL
	LD X
	SUB WIDTH
	JZ NLINE
	LD X
	ADD #1
	ST X
	JMP DRAW
NLINE:	LD ZERO
	ST X
	LD Y
	ADD #1
	ST Y
	JMP START
END:	HALT
.endcode
