.data
ZERO: DB #144, #0
COLOR: DB #145, #255
WIDTH: DB #146, #5
HEIGHT: DB #147, #3
Y: DB #148, #0
X: DB #149, #0
.enddata

.code
RIGHT:	POS
	LD ZERO
	LD2 COLOR
	LD3 ZERO
	PXL
	LD X
	SUB WIDTH
	JZ DOWN
	LD X
	ADD #1
	ST X
	LD2 Y
	JMP RIGHT
DOWN:	LD Y
	SUB HEIGHT
	JZ LEFT
	LD Y
	ADD #1
	ST Y
	LD X
	LD2 Y
	POS
	LD ZERO
	LD2 COLOR
	LD3 ZERO
	PXL
	JMP DOWN
LEFT:	LD X
	JZ UP
	SUB #1
	ST X
	LD X
	LD2 Y
	POS
	LD ZERO
	LD2 COLOR
	LD3 ZERO
	PXL
	JMP LEFT
UP:	LD Y
	JZ END
	SUB #1
	ST Y
	LD X
	LD2 Y
	POS
	LD ZERO
	LD2 COLOR
	LD3 ZERO
	PXL
	JMP UP
END:	HALT
.endcode
